import { useState } from 'react'
import { AvatarGroup } from './AvatarGroup'

# AvatarGroup

**Назначение.** Стопка перекрывающихся аватаров (с кольцом-обводкой и сворачиванием в «+N»).
Источник правды: Figma → ds-atoms → `avatars` (node `1:1782`). **Статус:** stable.

| prop    | значения              | по умолч. |
| ------- | --------------------- | --------- |
| `items` | `{name?, src?}[]`     | —         |
| `size`  | `number`              | `32`      |
| `max`   | `number` (видимых)    | —         |

## Живой вариант (добавляй/убирай участников)

export const ALL = ['Katya', 'Dog', 'Petya', 'Alan', 'Sara', 'Max', 'Lena']

export const Demo = () => {
  const [count, setCount] = useState(4)
  const items = ALL.slice(0, count).map((name) => ({ name }))
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
      <AvatarGroup items={items} max={4} size={36} />
      <div style={{ display: 'flex', gap: 8, alignItems: 'center' }}>
        <button onClick={() => setCount((c) => Math.max(1, c - 1))} style={{ width: 28, height: 28, borderRadius: 4, border: '1px solid var(--ds-color-lines)', cursor: 'pointer' }}>−</button>
        <button onClick={() => setCount((c) => Math.min(ALL.length, c + 1))} style={{ width: 28, height: 28, borderRadius: 4, border: '1px solid var(--ds-color-lines)', cursor: 'pointer' }}>+</button>
        <span style={{ fontSize: 13, opacity: 0.6 }}>участников: {count} (видимых 4, остальные → +N)</span>
      </div>
    </div>
  )
}
