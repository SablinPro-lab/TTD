import { useState } from 'react'
import { Avatar } from './Avatar'

# Avatar

**Назначение.** Круглый аватар пользователя: картинка или инициалы на сером фоне.
Источник правды: Figma → ds-atoms → `avatar` (node `1:1727`). **Статус:** stable.

| prop   | значения  | по умолч. | Figma                     |
| ------ | --------- | --------- | ------------------------- |
| `src`  | `string`  | —         | —                         |
| `name` | `string`  | —         | пресеты katya, dog, petya |
| `size` | `number`  | `40`      | —                         |

## Живой вариант (имя и размер)

export const NAMES = ['Katya', 'Petya', 'Dog', '']

export const Demo = () => {
  const [name, setName] = useState('Katya')
  const [size, setSize] = useState(48)
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
      <div style={{ display: 'flex', gap: 8 }}>
        {NAMES.map((n) => (
          <button key={n || 'empty'} onClick={() => setName(n)}
            style={{ padding: '4px 10px', borderRadius: 4, border: '1px solid var(--ds-color-lines)',
              background: name === n ? 'var(--ds-color-control-secondary)' : 'transparent', cursor: 'pointer', fontSize: 12 }}>
            {n || '(без имени)'}
          </button>
        ))}
      </div>
      <input type="range" min={24} max={96} value={size} onChange={(e) => setSize(Number(e.target.value))} style={{ maxWidth: 240 }} />
      <Avatar name={name} size={size} />
    </div>
  )
}

<Demo />
