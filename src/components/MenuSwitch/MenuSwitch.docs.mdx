## Живой вариант (кликни — выбор переключается)

export const ITEMS = ['All teams', 'All templates', 'Archive']

export const Demo = () => {
  const [active, setActive] = useState('All teams')
  return (
    <div style={{ display: 'flex', gap: 8, background: 'var(--ds-color-control-secondary)', padding: 8, borderRadius: 4 }}>
      {ITEMS.map((it) => (
        <MenuSwitch key={it} label={it} active={active === it} onClick={() => setActive(it)} />
      ))}
    </div>
  )
}
