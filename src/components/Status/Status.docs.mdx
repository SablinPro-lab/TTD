import { useState } from 'react'
import { Status } from './Status'

# Status

**Назначение.** Индикатор состояния: цветная точка + uppercase-подпись. Источник правды:
Figma → ds-atoms → `status` (node `1:1734`). **Статус:** stable.

| prop      | значения                              | по умолч. |
| --------- | ------------------------------------- | --------- |
| `variant` | `purple` · `green` · `red` · `stopped`| `green`   |
| `children`| `ReactNode` (текст подписи)           | —         |

## Живой вариант (выбери цвет)

export const LABELS = { purple: 'Rocket growth', green: 'On track', red: 'Failing', stopped: 'Stopped' }

export const Demo = () => {
  const [v, setV] = useState('green')
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
      <div style={{ display: 'flex', gap: 8, flexWrap: 'wrap' }}>
        {Object.keys(LABELS).map((k) => (
          <button key={k} onClick={() => setV(k)}
            style={{ padding: '4px 10px', borderRadius: 4, border: '1px solid var(--ds-color-lines)',
              background: v === k ? 'var(--ds-color-control-secondary)' : 'transparent', cursor: 'pointer', fontSize: 12 }}>
            {k}
          </button>
        ))}
      </div>
      <Status variant={v}>{LABELS[v]}</Status>
    </div>
  )
}
